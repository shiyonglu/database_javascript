CREATE TABLE professor(	
	ssn INTEGER,                
	profname VARCHAR(50),               
	status VARCHAR(10),                
	PRIMARY KEY(ssn)
);

CREATE TABLE course(	
	crscode CHAR(7),                
	crsname VARCHAR(100),               
	credit TINYINT,                
	PRIMARY KEY(crscode)
);

CREATE TABLE taught(			
	crscode CHAR(7),                	
	semester MEDIUMINT,               	
	ssn INTEGER,                	
	PRIMARY KEY(crscode, semester),	
	FOREIGN KEY(crscode) REFERENCES course(crscode),	FOREIGN KEY(ssn) REFERENCES professor(ssn)
);
